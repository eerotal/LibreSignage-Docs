128  $clash = FALSE;

131  foreach ($this->slides as $k => $s) {
132  if ($s->get_id() == $keep_id) {

135  $this->slides = array_values($this->slides);

138  }

142  }

144 
145  // Shift indices so that the index of $keep_id is left free.
146  $keep_i = $keep->get_index();
147  foreach ($this->slides as $k => $s) {
148  $s_i = $s->get_index();
149  $clash |= $s_i == $keep_i;
